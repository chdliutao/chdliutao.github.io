go基础数据结构实现--栈

Posted on 2019-10-24

package main

//go顺序栈实现

//SeqStack 顺序栈，使用数组构建
type SeqStack struct {
max int
data []int
top int
}

// New 栈的初始化操作
func (myStack *SeqStack) New(size int) {
myStack.data = make([]int, size)
myStack.max = size
myStack.top = -1
}

// Push 数据入栈操作
func (myStack *SeqStack) Push(value int) bool {
if myStack.top == myStack.max-1 {
return false
}
myStack.data[myStack.top] = value
myStack.top++
return true
}

// Pop 数据出栈操作
func (myStack *SeqStack) Pop() (int, bool) {
if myStack.top == -1 {
return 0, false
}
value := myStack.data[myStack.top]
myStack.top–
return value, true
}

func main() {
stack := SeqStack{}
stack.New(5)
}
